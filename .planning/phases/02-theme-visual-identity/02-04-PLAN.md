     const lum = ([r,g,b]) => 0.2126*lin(r)+0.7152*lin(g)+0.0722*lin(b);
     const ratio = (a,b) => { const [L1,L2]=[lum(hex2rgb(a)),lum(hex2rgb(b))].sort((x,y)=>y-x); return (L1+0.05)/(L2+0.05); };
     const r = ratio("#F4ECE0","#14110E");
